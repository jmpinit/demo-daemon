Demo-Daemon: API

setup()

description:

Called when code is loaded.

draw()

description:

Called every frame.

example: 
(function() {
	var d = 16;
	for(var y=0; y < canvas.height; y+=d) {
		for(var x=0; x < canvas.width; x+=d) {
			var c = (y^x)&255; if(c < 0) c=0; if(c > 255) c=255;
			ctx.fillStyle = 'rgb('+c+','+c+','+c+');';
			ctx.fillRect(x, y, d, d);
		}
	}
})

map2d(canvas, fn, dx=1, dy=1)

fn(x, y): [red, green, blue]

description:

Sets every canvas pixel to result of fn. Optionally draws with pixel size (dx, dy).
